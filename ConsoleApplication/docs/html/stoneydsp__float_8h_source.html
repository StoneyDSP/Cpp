178 inline Float& operator=(Float&& otherValue) noexcept
179 {
180 std::cout << this << " - " << typeid(*this).name() << " - Called Move Assignment Operator [=] from address " << &otherValue << " = " << otherValue.value;
181
182 if (this == &otherValue)
183 {
184 std::cout << " - Move failed!" << std::endl;
185 std::cout << std::endl;
186
187 return *this;
188 }
189
190 //* Performs no operation if you try to assign the object to itself. */
191 else //(&otherValue != this)
192 {
193 // Free the resource
194 value = 0.0F;
195
196 // Copy the data from the source object.
197 value = otherValue.value;
198
199 // Release the data from the source object so that
200 // the destructor does not free the memory multiple times.
201 otherValue.value = 0.0F;
202
203 std::cout << " - Move successfull!" << std::endl;
204 std::cout << std::endl;
205 }
206
207 return *this;
