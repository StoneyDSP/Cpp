183 {
184 std::cout << this << " - " << typeid(*this).name() << " - Called Move Assignment Operator [=] from address " << &otherValue << " = " << otherValue.value;
185

187 {

189 std::cout << std::endl;
190
191 return *this;
192 }
193
194 //* Performs no operation if you try to assign the object to itself. */
195 else //(&otherValue != this)
196 {
197 // Free the resource
198 value = Type(0.0);
199
200 // Copy the data from the source object.
201 value = otherValue.value;
202
203 // Release the data from the source object so that
204 // the destructor does not free the memory multiple times.
205 otherValue.value = Type(0.0);
206
207 std::cout << " - Move successfull!" << std::endl;
208 std::cout << std::endl;
